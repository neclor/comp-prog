\section{Specifications}\label{specifications}
%%%%%%%%%%%%%%%%%%%%%%%%

\boxed{
\begin{aligned}
&\text{create\_escale}: \mathbb{R} \times \mathbb{R} \times \text{String} \to Escale \\
&\text{pre}: \\
&\text{post}: \\
&\quad \text{escale\_x}(e) = x \\
&\quad \text{escale\_y}(e) = y \\
&\quad \text{escale\_name}(e) = n \\
&\quad \text{escale\_time}(e) = 0
\end{aligned}
}

\boxed{
\begin{aligned}
&\text{course\_create\_array}: Escale \times Escale \to Course \\
&\text{pre}: e_1 \neq e_2 \\
&\text{post}: \\
&\quad \text{course\_size}(C) = 2 \\
&\quad \text{course\_is\_circuit}(C) = \text{false} \\
&\quad \text{course\_at}(C, 0) = e_1 \\
&\quad \text{course\_at}(C, 1) = e_2 \\
&\quad \text{course\_time\_at}(C, 0) = 0
\end{aligned}
}

\boxed{
\begin{aligned}
&\text{course\_create\_list}: Escale \times Escale \to Course \\
&\text{pre}: e_1 \neq e_2 \\
&\text{post}: \\
&\quad \text{course\_size}(C) = 2 \\
&\quad \text{course\_head}(C) = e_1 \\
&\quad \text{course\_tail}(C) = e_2 \\
&\quad \text{course\_time\_at}(C, 0) = 0
\end{aligned}
}

\boxed{
\begin{aligned}
&\text{course\_append}: \Course \times \Escale \to \Course \\
&\text{pre}: \text{course\_size}(C) \geq 2 \\
&\text{post}: \\
&\quad \text{course\_size}(C') = \text{course\_size}(C) + 1 \\
&\quad \text{course\_at}(C', \text{course\_size}(C)) = e \\
&\quad \forall i \in [0, \text{course\_size}(C)-1], \text{course\_at}(C', i) = \text{course\_at}(C, i)
\end{aligned}
}

\boxed{
\begin{aligned}
&\text{course\_pop}: \Course \to \Course \\
&\text{pre}: \text{course\_size}(C) > 2 \\
&\text{post}: \\
&\quad \text{course\_size}(C') = \text{course\_size}(C) - 1 \\
&\quad \forall i \in [0, \text{course\_size}(C')-1], \text{course\_at}(C', i) = \text{course\_at}(C, i)
\end{aligned}
}

\boxed{
\begin{aligned}
&\text{course\_is\_circuit}: \Course \to \mathbb{B} \\
&\text{post}: \\
&\quad \text{result} = (\text{course\_at}(C, 0) = \text{course\_at}(C, \text{course\_size}(C)-1))
\end{aligned}
}

\boxed{
\begin{aligned}
&\text{course\_total\_time}: \Course \to \mathbb{R} \\
&\text{post}: \\
&\quad \text{result} = \sum_{i=0}^{\text{course\_size}(C)-1} \text{escale\_time}(\text{course\_at}(C, i))
\end{aligned}
}
